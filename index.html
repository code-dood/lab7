import hashlib
import time

class Block:

    def __init__(self, index, previous_hash, data):

        self.index = index
        self.timestamp = int(time.time())
        self.previous_hash = previous_hash
        self.data = data
        self.hash = self.calculate_hash()

    def calculate_hash(self):

        text = f"{self.index}{self.timestamp}{self.previous_hash}{self.data}"
        return hashlib.sha256(text.encode()).hexdigest()

if __name__ == "__main__":

    print("Timestamp:", block.timestamp)
    print("Previous Hash:", block.previous_hash)
    print("Data:", block.data)
    print("Hash:", block.hash)

2. IMPLEMENTATION OF BLOCKCHAIN


import hashlib
import time

class Block:

    def __init__(self, index, previous_hash, data):

        self.index = index
        self.timestamp = int(time.time())
        self.previous_hash = previous_hash
        self.data = data
        self.hash = self.calculate_hash()

    def calculate_hash(self):

        text = f"{self.index}{self.timestamp}{self.previous_hash}{self.data}"
        return hashlib.sha256(text.encode()).hexdigest()

class Blockchain:

    def __init__(self):

        self.chain = [self.create_genesis_block()]

    def create_genesis_block(self):

        return Block(0, "0", "Genesis Block")

    def get_latest_block(self):

        return self.chain[-1]

    def add_block(self, data):

        previous = self.get_latest_block()

        new_block = Block(len(self.chain), previous.hash, data)

        self.chain.append(new_block)

if __name__ == "__main__":

    blockchain = Blockchain()

    blockchain.add_block("First block")
    blockchain.add_block("Second block")
    blockchain.add_block("Third block")

    for block in blockchain.chain:

        print(f"\n--- Block {block.index} ---")

        print("Timestamp:", block.timestamp)
        print("Previous Hash:", block.previous_hash)
        print("Hash:", block.hash)
        print("Data:", block.data)

4. SHOR'S ALGORITHM


import math

N = 15

for a in range(2, N):

    if math.gcd(a, N) != 1:
        continue

    r = 1

    while pow(a, r, N) != 1:
        r += 1

    if r % 2 == 0:

        p = math.gcd(pow(a, r//2) - 1, N)
        q = math.gcd(pow(a, r//2) + 1, N)

        if p > 1 and q > 1:

            print("Factors of", N, "are:", p, "and", q)
            break

import secrets
import hashlib
import time

class CryptoWallet:

    def __init__(self, initial_balance=1000):

        self.private_key = secrets.token_hex(32)

        self.public_key = hashlib.sha256(self.private_key.encode()).hexdigest()

        self.address = "0x" + hashlib.sha256(self.public_key.encode()).hexdigest()[-40:]

        self.balance = initial_balance

def apply_unitary(matrix,state):

    result=[0,0]

    result[0]=matrix[0][0]*state[0] + matrix[0][1]*state[1]
    result[1]=matrix[1][0]*state[0] + matrix[1][1]*state[1]

    return result

11. INNER PRODUCT


def complex_inner_product(v1,v2):

    if len(v1)!=len(v2):

        return "Vectors must be same length"

    result=0

    for i in range(len(v1)):

        result += complex(v1[i]).conjugate() * complex(v2[i])

    return result

12. HADAMARD GATE


import math

def hadamard(qubit):

    factor = 1/math.sqrt(2)

    H = [
        [factor,factor],
        [factor,-factor]
    ]

    result = [
        H[0][0]*qubit[0] + H[0][1]*qubit[1],
        H[1][0]*qubit[0] + H[1][1]*qubit[1]
    ]

    return result
